#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, head, total = 0;
    printf("Enter number of requests: ");
    scanf("%d", &n);

    int req[n], done[n]; //done[]=flags whether that request is already processed (0=not done,1 = done)
    printf("Enter request sequence:\n");
    for (int i = 0; i < n; i++) {
        scanf("%d", &req[i]);
        done[i] = 0; // mark as not served
    }

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("\nHead movement: %d", head);

    for (int count = 0; count < n; count++) {
        int min = 9999, pos = -1;

        // find the nearest request
        for (int i = 0; i < n; i++) {
            if (!done[i]) {
                int diff = abs(req[i] - head);
                if (diff < min) {
                    min = diff;
                    pos = i;
                }
            }
        }

        total += min;
        head = req[pos];
        done[pos] = 1;

        printf(" -> %d", head);
    }

    printf("\n\nTotal head movement = %d", total);
    printf("\nAverage seek time = %.2f\n", (float)total / n);

    return 0;
}
