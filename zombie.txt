#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>
#include <stdlib.h>

// Simple bubble sort
void bubbleSort(int a[], int n) {
    int i, j, temp;
    for (i = 0; i < n - 1; i++) //for every i, every element in arr in checkee and compared
        for (j = 0; j < n - i - 1; j++)
            if (a[j] > a[j + 1]) {
                temp = a[j];
                a[j] = a[j + 1];
                a[j + 1] = temp;
            }
}

//A Zombie process is a process that has finished executing (it called exit()),
//but its parent process has not yet collected its exit status using wait() or waitpid().
//So, the child is dead, but its entry in the process table is still there — just waiting for the parent to acknowledge it.
//Zombie = "dead process whose ghost still remains in the process table."

int main() {
    int n;
    printf("Enter number of integers: ");
    scanf("%d", &n);

    int arr[n];
    printf("Enter %d integers: ", n);
    for (int i = 0; i < n; i++)
        scanf("%d", &arr[i]);

    pid_t pid = fork(); //fork() is a system call that creates a new process.

    if (pid == 0) {
        // Child process
        printf("\n[Child] PID=%d, Parent PID=%d\n", getpid(), getppid());
        bubbleSort(arr, n);
        printf("[Child] Sorted array: ");
        for (int i = 0; i < n; i++)
            printf("%d ", arr[i]);
        printf("\n");
	
	exit(0); //Child process exits immediately after printing —
		//This is where the Zombie state begins (until the parent collects its exit status).
    } 
    else if (pid > 0) {
        // Parent process
        printf("\n[Parent] PID=%d, Child PID=%d\n", getpid(), pid);
        bubbleSort(arr, n);
        printf("[Parent] Sorted array: ");
        for (int i = 0; i < n; i++)
            printf("%d ", arr[i]);
        printf("\n");

        printf("[Parent] Sleeping for 10 seconds (child becomes zombie)...\n");
        sleep(10);
    } 
    else {
        perror("fork failed");
    }

    return 0;
}
