#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[]) {
    printf("\n[Child Program] Received %d arguments\n", argc - 1);

    if (argc < 2) {
        printf("No array elements received!\n");
        return 0;
    }

    int n = argc - 1;
    int arr[n];

    // Convert command-line strings back to integers
    for (int i = 0; i < n; i++) 
        arr[i] = atoi(argv[i + 1]);

    printf("[Child Program] Array in reverse order: ");
    for (int i = n - 1; i >= 0; i--)
        printf("%d ", arr[i]);
    printf("\n");
	
	return 0;
}
